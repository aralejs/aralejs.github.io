遮罩会根据 DOM 的宽高进行调节

            var shim2 = new Shim('#example2 .overlay').sync();

            $('#example2 input').blur(function(e){
                var value = e.currentTarget.value;

                $('#example2 .overlay').css({
                    'height': value,
                    'width': value
                });

                shim2.sync();
            });
#example2 .overlay
